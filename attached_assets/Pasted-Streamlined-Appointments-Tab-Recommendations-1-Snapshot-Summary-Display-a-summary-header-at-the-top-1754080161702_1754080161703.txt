import { Button, Avatar, Chip } from "@radix-ui/react-xyz";

function AppointmentCard({
  time,
  clientName,
  clientInitials,
  clientPhone,
  location,
  sessionType = "Individual Counseling",
  isEditedType,
  onPrep,
  onNote,
  onCheckIn,
  onEdit,
  onDelete,
  ...props
}) {
  return (
    <div className="appointment-card">
      <div className="card-header">
        <span className="time">{time}</span>
        <Avatar>{clientInitials}</Avatar>
        <span className="client-name">{clientName}</span>
        {clientPhone && (
          <a href={`tel:${clientPhone}`} title="Call client" className="phone-icon">
            üìû
          </a>
        )}
        <span className="session-type">
          {isEditedType ? <Chip>{sessionType}</Chip> : sessionType}
        </span>
        <span className="location">{location}</span>
      </div>
      <div className="card-actions">
        <Button onClick={onPrep}>Session Prep</Button>
        <Button onClick={onNote}>Note</Button>
        <Button onClick={onCheckIn}>Check-In</Button>
        <Button onClick={onEdit}>Edit</Button>
        <Button onClick={onDelete} color="red">Delete</Button>
      </div>
      {/* Optional expanded content */}
      {props.expanded && (
        <div className="card-expanded">
          {/* Insert session prep, progress note summary, communication, etc here */}
        </div>
      )}
    </div>
  );
}

export default AppointmentCard;
